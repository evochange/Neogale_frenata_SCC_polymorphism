### NGSadmix analysis

# 1. CONDUCT SNP CALLS BASED ON GENOTYPE LIKELIHOODS
# 2. FILTER OUTPUT FOR DNA DAMAGE
# 3. FILTER SNPS DOR INDEPENDENCE - DISTANCE BASED
# For these first 3 steps, follow the PCA pipeline

# 4. CALCULATE GENOTYPE LIKELIHOODS FOR INFERRED SNPS:
# The flag -doGlf will give the genotype likelihoods output.

angsd -b samples_frenata.txt -ref GCF_020171115.1_ASM_NN_V1_genomic.fna -out gl_min22_mincov22_maxcov132_maf0.020.damfilt_20K -sites ../PCA/snps_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt -rf ../PCA/chr_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt -uniqueOnly 1 -remove_bads 1 -minMapQ 20 -minQ 20 -GL 1 -doCounts 1 -doMajorMinor 1 -minMaf 0.020 -SNP_pval 1e-6 -doMaf 1 -doGlf 2


# 5. RUN NGSadmix:
# do multiple repetitions per K (eg. 1000)
for R in {1..1000};
do NGSadmix -likes ../gl_min22_mincov22_maxcov132_maf0.020.damfilt_20K.beagle.gz -K 2 -outfiles gl_min22_mincov22_maxcov132_maf0.020.damfilt_20K_rep$R -P 2;
done
