## PCA - IBS APPROACH (ANGSD) ----

#NOTES:
# For the IBS approach, info here: http://www.popgen.dk/angsd/index.php/PCA_MDS
# For other approaches, info here: http://www.popgen.dk/angsd/index.php/PCA


##1. CONDUCT SNP CALLS BASED ON GENOTYPE LIKELIHOODS:
#Output results to Plink format (-doPlink 2) so that we can filter sites for independence with plink. Info here: http://www.popgen.dk/angsd/index.php/Plink. 

angsd -P 4 -b samples_frenata.txt -ref GCF_020171115.1_ASM_NN_V1_genomic.fna -out snps_frenata_min22_maxcov132_maf0.020 -uniqueOnly 1 -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 22 -setMaxDepth 132 -doCounts 1 -GL 1 -doMajorMinor 1 -doMaf 1 -minMaf 0.020 -SNP_pval 1e-6 -doGeno -4 -doPost 1 -doPlink 2

#PCA for local transition regions

angsd -P 4 -b samples_east_frenata.txt -ref GCF_020171115.1_ASM_NN_V1_genomic.fna -out snps_east_min11_maxcov66_maf0.040 -uniqueOnly 1 -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 11 -setMaxDepth 66 -doCounts 1 -GL 1 -doMajorMinor 1 -doMaf 1 -minMaf 0.040 -SNP_pval 1e-6 -doGeno -4 -doPost 1 -doPlink 2
angsd -P 4 -b samples_west_frenata.txt -ref GCF_020171115.1_ASM_NN_V1_genomic.fna -out snps_west_min11_maxcov66_maf0.040 -uniqueOnly 1 -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 11 -setMaxDepth 66 -doCounts 1 -GL 1 -doMajorMinor 1 -doMaf 1 -minMaf 0.040 -SNP_pval 1e-6 -doGeno -4 -doPost 1 -doPlink 2

#PCA for MC1R genomic region

angsd -P 4 -b ../allpops_MC1R.txt -ref GCF_020171115.1_ASM_NN_V1_genomic.fna -r 337:139300-170400 -out snps_allpops_min22_maxcov132_maf0.020_MC1R -uniqueOnly 1 -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 11 -setMaxDepth 132 -doCounts 1 -GL 1 -doMajorMinor 1 -doMaf 1 -minMaf 0.020 -SNP_pval 1e-6 -doGeno -4 -doPost 1 -doPlink 2


#2. FILTER OUTPUT FOR DNA DAMAGE
#Use the list of sites with potential damage identified in folder 1.

awk 'NR==FNR{a[$1,$2];next} !(($1,$4) in a)' ../DNA_damage/list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt snps_frenata_min22_maxcov132_maf0.020.tped > snps_frenata_min22_maxcov132_maf0.020.damfilt.tped

#PCA for local transition regions

awk 'NR==FNR{a[$1,$2];next} !(($1,$4) in a)' ../DNA_damage/list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt snps_east_min11_maxcov66_maf0.040.tped > snps_east_min11_maxcov66_maf0.040.damfilt.tped
awk 'NR==FNR{a[$1,$2];next} !(($1,$4) in a)' ../DNA_damage/list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt snps_west_min11_maxcov66_maf0.040.tped > snps_west_min11_maxcov66_maf0.040.damfilt.tped

#PCA for MC1R genomic region

awk 'NR==FNR{a[$1,$2];next} !(($1,$4) in a)' ../DNA_damage/list-sites2remove-frenata_final_min11_nivalis_freq0.020-GL.txt snps_allpops_min22_maxcov132_maf0.020_MC1R.tped > snps_allpops_min22_maxcov132_maf0.025_MC1R.damfilt.tped

#Create a tfam plink file for the DNA damage filtered tped file

cp snps_frenata_min22_maxcov132_maf0.020.tfam snps_frenata_min22_maxcov132_maf0.020.damfilt.tfam

#PCA for local transition regions

cp snps_east_min11_maxcov66_maf0.040.tfam snps_east_min11_maxcov66_maf0.040.damfilt.tfam
cp snps_west_min11_maxcov66_maf0.040.tfam snps_west_min11_maxcov66_maf0.040.damfilt.tfam

#PCA for MC1R genomic region

cp snps_allpops_min22_maxcov132_maf0.020_MC1R.tfam snps_allpops_min22_maxcov132_maf0.020_MC1R.damfilt.tfam 


#3. FILTER SNPS FOR INDEPENDENCE - DISTANCE BASED
#3a. Produce bed files (all files/different windows)

plink --tfile snps_frenata_min22_maxcov132_maf0.020.damfilt --make-bed --bp-space 20000 --allow-no-sex --allow-extra-chr --out snps_frenata_min22_maxcov132_maf0.020.damfilt.20k

#PCA for local transition regions

plink --tfile snps_east_min11_maxcov66_maf0.040.damfilt --make-bed --bp-space 20000 --allow-no-sex --allow-extra-chr --out snps_east_min11_maxcov66_maf0.040.damfilt.20K
plink --tfile snps_west_min11_maxcov66_maf0.040.damfilt --make-bed --bp-space 20000 --allow-no-sex --allow-extra-chr --out snps_west_min11_maxcov66_maf0.040.damfilt.20k

#PCA for MC1R genomic region

plink --tfile snps_allpops_min22_maxcov132_maf0.025_MC1R.damfilt --make-bed --allow-no-sex --allow-extra-chr --out snps_allpops_min22_maxcov132_maf0.020_MC1R.damfilt

#3b. Produce site files from bed:

awk '{print $1, $4}' snps_frenata_min22_maxcov132_maf0.020.damfilt.20k.bim > snps_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt

#PCA for local transition regions

awk '{print $1, $4}' snps_east_min11_maxcov66_maf0.040.damfilt.20K.bim > snps_east_min11_maxcov66_maf0.040.damfilt.20K.txt
awk '{print $1, $4}' snps_west_min11_maxcov66_maf0.040.damfilt.20k.bim > snps_west_min11_maxcov66_maf0.040.damfilt.20K.txt

#PCA for MC1R genomic region

awk '{print $1, $4}' snps_allpops_min22_maxcov132_maf0.025_MC1R.damfilt.bim > snps_allpops_min22_maxcov132_maf0.020_MC1R.damfilt.txt

#3c.Produce chr file from sites file

awk '{print $1}' snps_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt |sort|uniq > chr_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt

#PCA for local transition regions

awk '{print $1}' snps_east_min11_maxcov66_maf0.040.damfilt.20K.txt |sort|uniq > chr_snps_min11_maxcov66_maf0.040.damfilt.20K.txt
awk '{print $1}' snps_west_min11_maxcov66_maf0.040.damfilt.20K.txt |sort|uniq > chr_west_min11_maxcov66_maf0.040.damfilt.20K.txt

#PCA for MC1R genomic region

awk '{print $1}' snps_allpops_min22_maxcov132_maf0.025_MC1R.damfilt.txt |sort|uniq > chr_allpops_min22_maxcov132_maf0.020_MC1R.damfilt.txt
	
#3c. Index sites files

angsd sites index snps_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt

#PCA for local transition regions

angsd sites index snps_east_min11_maxcov66_maf0.040.damfilt.20K.txt
angsd sites index snps_west_min11_maxcov66_maf0.040.damfilt.20K.txt

#PCA for MC1R genomic region

angsd sites index snps_allpops_min22_maxcov132_maf0.020_MC1R.damfilt.txt


#4. CALCULATE PCA - IBS APPROACH:
#Sites file comes from the previous analyses
#-rf flag will only analyse the scaffolds included in the chr file, which will reduce the time of the run.

angsd -b ../0_merged/samples_frenata.txt -doIBS 2 -doCounts 1 -sites snps_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt -rf chr_frenata_min22_maxcov132_maf0.020.damfilt.20k.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -doMajorMinor -1 -GL 1 -doCov 1 -out ibs_frenata_min22_maxcov132_maf0.020.damfilt.20k -P 6

#PCA for local transition regions

angsd -b ../0_merged/samples_east_frenata.txt -doIBS 2 -doCounts 1 -sites snps_east_min11_maxcov66_maf0.040.damfilt.20K.txt -rf chr_snps_min11_maxcov66_maf0.040.damfilt.20K.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -doMajorMinor -1 -GL 1 -doCov 1 -out ibs_east_min11_maxcov66_maf0.040.damfilt.20K -P 4
angsd -b ../0_merged/samples_west_frenata.txt -doIBS 2 -doCounts 1 -sites snps_west_min11_maxcov66_maf0.040.damfilt.20K.txt -rf chr_west_min11_maxcov66_maf0.040.damfilt.20K.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -doMajorMinor -1 -GL 1 -doCov 1 -out ibs_west_min11_maxcov66_maf0.040.damfilt.20K -P 4

#PCA for MC1R genomic region

angsd -b ../allpops_MC1R.txt -doIBS 2 -doCounts 1 -sites snps_allpops_min22_maxcov132_maf0.020_MC1R.damfilt.txt -rf chr_allpops_min22_maxcov132_maf0.020_MC1R.damfilt.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -doMajorMinor -1 -GL 1 -doCov 1 -out ibs_allpops_min22_maxcov132_maf0.020_MC1R.damfilt -P 6


#5. PLOT WITH R:
#With the .covMat output
#5a. Modify the .covMat output to include sample codes and coat colour, and header with PC components names

name <- "angsdput.covMat"
m <- as.matrix(read.table(name))
e <- eigen(m)
plot(e$vectors[,1:2],lwd=2,ylab="PC 2",xlab="PC 1",main="Principal components",col=rep(1:22),pch=16)
