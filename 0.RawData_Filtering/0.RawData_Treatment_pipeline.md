## Raw data Treatment

#0. INDEX REFERENCE GENOME
#Prepare the reference genome for all analyses; all three indexings are needed for this pipeline.

	0.1. INDEX REFERENCE WITH BWA:
	bwa index GCF_020171115.1_ASM_NN_V1_genomic.fna
	
	0.2. INDEX REFERENCE WITH PICARD:
	java -jar /usr/local/bin/picard.jar CreateSequenceDictionary R=GCF_020171115.1_ASM_NN_V1_genomic.fna O=GCF_020171115.1_ASM_NN_V1_genomic.dict
	
	0.3. INDEX REFERENCE WITH SAMTOOLS:
	samtools faidx GCF_020171115.1_ASM_NN_V1_genomic.fna


##1. QUALITY AND ADAPTER TRIMMING WITH TRIMMOMATIC:
	
	java -jar /usr/local/bin/Trimmomatic-0.38/trimmomatic-0.38.jar PE -threads 8 $1_*_L*_1.fq.gz $1_*_L*_2.fq.gz $1_1P.fastq.gz $1_1U.fastq.gz $1_2P.fastq.gz $1_2U.fastq.gz ILLUMINACLIP:/home/.../adapters/TruSeq3-PE-2.fa:2:30:10:2 TRAILING:15 SLIDINGWINDOW:5:15 MINLEN:25


##2. MERGE OVERLAPPING PAIRS WITH PEAR:
	
	pear -f $1_1P.fastq.gz -r $1_2P.fastq.gz -o	$1_OP -n 50 -j 10 -v 10
	gzip $1*fastq
	cp $1_OP.unassembled.forward.fastq.gz ../03_cleaned-data/$1_R1.fastq.gz
	cp $1_OP.unassembled.reverse.fastq.gz ../03_cleaned-data/$1_R2.fastq.gz
	cat $1_OP.assembled.fastq.gz ../01_trim/hiseq-novogene/$1_1U.fastq.gz ../01_trim/hiseq-novogene/$1_2U.fastq.gz > ../03_cleaned-data/$1_S.fastq.gz


##3. MAPPING TO SORT WITH BWA-MEM AND SAMTOOLS:
	
	bwa mem -M -t 4 GCF_020171115.1_ASM_NN_V1_genomic.fna $1_R1.fastq.gz $1_R2.fastq.gz > $1_PE.assembly50k.sam
	samtools view -@ 5 -Sb $1_PE.assembly50k.sam > $1_PE.assembly50k.bam
	rm $1_PE.assembly50k.sam
	samtools sort -@ 5 -o $1_PE.assembly50k.sorted.bam $1_PE.assembly50k.bam
	rm $1_PE.assembly50k.bam

	bwa mem -M -t 8 GCF_020171115.1_ASM_NN_V1_genomic.fna $1_S.fastq.gz > $1_SE.assembly50k.sam
	samtools view -@ 7 -Sb $1_SE.assembly50k.sam > $1_SE.assembly50k.bam
	rm $1_SE.assembly50k.sam
	samtools sort -@ 7 -o $1_SE.assembly50k.sorted.bam $1_SE.assembly50k.bam
	rm $1_SE.assembly50k.bam
	

##4. CREATE READ GROUPS WITH PICARD:
	
	cat east_frenata_L1.txt | xargs -n 1 -P 8 sh -c 'java -Xmx2g -jar picard.jar AddOrReplaceReadGroups I=../04_mapping/$0_L1_PE.vison.sorted.bam O=$0_L1_PE.vison.rg.bam RGID=run01-lane1 RGLB=$0-lib1 RGPL=illumina RGSM=$0 RGPU=run01-lane1' &> log-rg-PE-L1_east.vison.txt


##5. MERGE .BAM FILES FROM DIFFERENT RUNS WITH SAMTOOLS: (merge_bams.py)

	cat frenata_east.txt | xargs -n 1 -P 10 sh -c 'echo samtools merge $0_merged.vision.bam ../05_picard/$0_*.vision.rg.bam' &> log_samtools_east_merge_museum.txt


##6. REMOVE DUPLICATES WITH PICARD: (remove_dup.py)

	cat ../samples_frenata.txt | xargs -n 1 -P 4 sh -c 'java -Xmx6g -jar picard.jar MarkDuplicates INPUT=../06_samtools/$0_merged.vision.bam OUTPUT=$0.vison.rmdup.bam METRICS_FILE=$0.vison.metrics.txt REMOVE_DUPLICATES=true ASSUME_SORTED=true MAX_FILE_HANDLES_FOR_READ_ENDS_MAP=1000 MAX_RECORDS_IN_RAM=100000000 VALIDATION_STRINGENCY=LENIENT' &> log_Picard_duplicates_vison_museum.txt
		
##7. INDEX BAM WITH SAMTOOLS: (index_bam.py)

	cat ../samples_frenata.txt | xargs -n 1 -P 10 sh -c 'echo samtools index 	$0.vison.rmdup.bam 	$0.vison.rmdup.bai' &> log_GATK_museum_vison.txt
	 
##8. INDEL COORDINATES WITH GATK: (indel_coord.py)

	cat ../samples_frenata.txt | xargs -n 1 -P 5 sh -c 'java -Xmx6g -jar GenomeAnalysisTK.jar -T RealignerTargetCreator -R GCF_020171115.1_ASM_NN_V1_genomic.fna -I $0.vison.rmdup.bam -o $0.vison.realigner.intervals' &> log_GATK_indels_museum_vison.txt

##9. INDEL REALIGNMENT WITH GATK: (indel_realign.py)

	cat ../frenata_east_samples.txt | xargs -n 1 -P 6 sh -c 'java -Xmx6g -jar GenomeAnalysisTK.jar -T IndelRealigner -R GCF_020171115.1_ASM_NN_V1_genomic.fna -I $0.vison.rmdup.bam -targetIntervals $0.vison.realigner.intervals -o $0.vison.real.bam' &> log_final_BAM_museum.txt
