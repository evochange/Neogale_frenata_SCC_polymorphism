## Filtering WGS dataset from C->T and G->A mutations


## 1. Produce allele frequencies estimates for the complete dataset, for sites with mininum 11 individuals (out of 44):
## Force reference as major (-doMajorMinor 4) as it will be the only way to distinguish C->T from T->C in angsd output

angsd -doMajorMinor 1 -doMaf 1 -SNP_pval 1e-6 -GL 1 -ref ./PATH/Neogale_vison/GCF_020171115.1_ASM_NN_V1_genomic.fna -bam samples_frenata_bams.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 11 -P 10 -out frenata_allpops_domajor1_min11_vison

## 2. Export lines where C is major and T is minor, or G is major and A is minor, and the minor allele frequency is >= 0.0227 (2/88 = 0.022727) -> 88 genotypes in total

zcat frenata_allpops_domajor1_min11_vison.mafs.gz | awk '{if (($3=="C" && $4=="T" && $6>=0.020) || ($3=="G" && $4=="A" && $6>=0.020)) print}' > sites2remove-frenata_final_min11_vison_freq0.020-GL.txt


## 3. Get list of sites to remove from the dataset:

awk '{print $1"\t"$2}' sites2remove-frenata_final_min11_vison_freq0.020-GL.txt > list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt


## 4. Use the list to filter out sites from thefinal output
## a) Filter DNA damage sites from list of sites to consider for WG scans (minimum of 5 individuals in each population in consideration in the scan):

awk 'NR==FNR{a[$1,$2];next} !(($1,$2) in a)' list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt ./filter_min5/all_pops/frenata_complete_vison_min5-sites.txt > frenata_final_all_pops_vison_min5-sites.damfilt.txt
awk 'NR==FNR{a[$1,$2];next} !(($1,$2) in a)' list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt ./filter_min5/east/frenata_east_vison_min5-sites.txt > frenata_final_east_vison_min5-sites.damfilt.txt
awk 'NR==FNR{a[$1,$2];next} !(($1,$2) in a)' list-sites2remove-frenata_final_min11_vison_freq0.020-GL.txt ./filter_min5/west/frenata_west_vison_min5-sites.txt > frenata_final_west_vison_min5-sites.damfilt.txt
