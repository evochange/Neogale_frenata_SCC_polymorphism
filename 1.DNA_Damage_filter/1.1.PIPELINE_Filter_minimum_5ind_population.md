## FILTER SITES BY MINIMUM CONTRIBUTION 5 PER POPULATION

##1. CALCULATE allele frequency likelihoods for sites covered by at least six individuals per pop. Calculate sites with minimum representation of 5 specimens in each morph/geography combination

angsd -doMajorMinor 1 -doMaf 1 -GL 1 -ref ./PATH/GCF_020171115.1_ASM_NN_V1_genomic.fna -bam East_brown_samples.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 5 -P 6 -out brown_east_vison_min5
angsd -doMajorMinor 1 -doMaf 1 -GL 1 -ref ./PATH/GCF_020171115.1_ASM_NN_V1_genomic.fna -bam East_white_samples.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 5 -P 6 -out white_east_vison_min5
angsd -doMajorMinor 1 -doMaf 1 -GL 1 -ref ./PATH/GCF_020171115.1_ASM_NN_V1_genomic.fna -bam West_brown_samples.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 5 -P 6 -out brown_west_vison_min5
angsd -doMajorMinor 1 -doMaf 1 -GL 1 -ref ./PATH/GCF_020171115.1_ASM_NN_V1_genomic.fna -bam West_white_samples.txt -uniqueOnly -remove_bads 1 -minQ 20 -minMapQ 20 -minInd 5 -P 6 -out white_west_vison_min5

##2. Produce -sites file with common sites for each comparison. Extract site coordinates from mafs estimations above

zcat brown_east_vison_min5.mafs.gz | awk '{ print $1"\t"$2 }' > brown_east_vison_min5-sites.txt
zcat white_east_vison_min5.mafs.gz | awk '{ print $1"\t"$2 }' > white_east_vison_min5-sites.txt
zcat brown_west_vison_min5.mafs.gz | awk '{ print $1"\t"$2 }' > brown_west_vison_min5-sites.txt
zcat white_west_vison_min5.mafs.gz | awk '{ print $1"\t"$2 }' > white_west_vison_min5-sites.txt

##3. Create scaffold list

awk '{ print $1 }' brown_east_vison_min5-sites.txt > scaffold_list_vison.txt
uniq scaffold_list_vison.txt > uniq_list.txt
rm scaffold_list_vison.txt
mv uniq_list.txt ./scaffold_list_vison.txt

##4. Create empty .txt file for each scaffold per combination of morph/geography

cat scaffold_list_vison.txt | xargs -n 1 -P 20 sh -c ' touch $0_brown_east_vison_min5-sites.txt'
cat scaffold_list_vison.txt | xargs -n 1 -P 20 sh -c ' touch $0_white_east_vison_min5-sites.txt'
cat scaffold_list_vison.txt | xargs -n 1 -P 20 sh -c ' touch $0_brown_west_vison_min5-sites.txt'
cat scaffold_list_vison.txt | xargs -n 1 -P 20 sh -c ' touch $0_white_west_vison_min5-sites.txt'

##5. Split sites in sites list per scaffold (inputs sites to the files created immediatly before)

awk '{print>$1"_brown_east_vison_min5-sites.txt"}' brown_east_vison_min5-sites.txt
awk '{print>$1"_white_east_vison_min5-sites.txt"}' white_east_vison_min5-sites.txt
awk '{print>$1"_brown_west_vison_min5-sites.txt"}' brown_west_vison_min5-sites.txt
awk '{print>$1"_white_west_vison_min5-sites.txt"}' white_west_vison_min5-sites.txt

##6. Compares brown/white files for each scaffold and outputs common sites to a new unique file

grep -xiFf

awk 'NR==FNR{a[$1,$2];next} ($1,$2) in a{print $0, a[$1,$2]}' $1_brown_east_vison_min5-sites.txt $1_white_east_vison_min5-sites.txt >> frenata_east_vison_min5-sites.txt
awk 'NR==FNR{a[$1,$2];next} ($1,$2) in a{print $0, a[$1,$2]}' $1_brown_west_vison_min5-sites.txt $1_white_west_vison_min5-sites.txt >> frenata_west_vison_min5-sites.txt

##7. FOR THE COMPLETE DATASET

cat scaffold_list_vison.txt | xargs -n 1 -P 10 sh -c ' touch $0_fr_east.txt'
cat scaffold_list_vison.txt | xargs -n 1 -P 10 sh -c ' touch $0_fr-west.txt'
